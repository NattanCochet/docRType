46void Systems::shootHandling(
47 CreateEntity &ce, Registry &r, VirtualKeyBoard &keyboard, Position &currentPosition,
48 Clock &clocks, const std::size_t &index
49)
50{
51 static constexpr float SPAM_PROTECTION_DELAY = 0.1f;
52 static constexpr float CHARGE_THRESHOLD = 0.3f;
53 static constexpr float MAX_CHARGE_TIME = 1.1f;
54 static constexpr float BASE_DAMAGE = 80.0f;
55
56 sf::Clock &keyPressing = clocks.getClock("keyPressing");
57 sf::Clock &spamProtection = clocks.getClock("spamProtection");
58
59 if (spamProtection.getElapsedTime().asSeconds() < SPAM_PROTECTION_DELAY) {
60 return;
61 }
62
63 if (keyboard.isKeyIsPressed(VirtualKeyBoard::CONTROL::SHOOT)) {
64 if (!keyboard.isCharging()) {
65 keyboard.setIsCharging(true);
66 keyPressing.restart();
67 }
68 return;
69 }
70
71 if (!keyboard.isCharging()) {
72 return;
73 }
74
75 keyboard.setIsCharging(false);
76 const float chargeTime = keyPressing.getElapsedTime().asSeconds();
77
78 if (chargeTime >= CHARGE_THRESHOLD) {
79 const float clampedTime = std::min(chargeTime, MAX_CHARGE_TIME);
80 const float damage = BASE_DAMAGE * (clampedTime / MAX_CHARGE_TIME);
81 ce.createBeam(currentPosition.getPosition(), r, damage, index);
82 } else {
83 ce.createLaser(currentPosition.getPosition(), r, index);
84 }
85
86 spamProtection.restart();
87}

89void Systems::movePlayerFromControllable(
90 sf::Vector2f &pos, const Hitbox &hitbox, VirtualKeyBoard &keyboard, Clock &clock,
91 const std::pair<std::size_t, std::size_t> &sizeWindow
92)
93{
94 sf::Clock &movePlayer = clock.getClock("movePlayer");
95 if (movePlayer.getElapsedTime().asMilliseconds() <= 16) {
96 return;
97 }
98
99 static const float speedValue = 5; // a changer en fonction de la vitesse
100
101 float newPosX = 0.0f;
102 float newPosY = 0.0f;
103
104 if (keyboard.isKeyIsPressed(VirtualKeyBoard::CONTROL::DOWN)) {
105 if (!hitbox.isCollisionInDirection(Hitbox::DIRECTION::DOWN) &&
106 pos.y + hitbox.getSize()[1] <= sizeWindow.second) {
107 newPosY += speedValue;
108 }
109 }
110
111 if (keyboard.isKeyIsPressed(VirtualKeyBoard::CONTROL::RIGHT)) {
112 if (!hitbox.isCollisionInDirection(Hitbox::DIRECTION::RIGHT) &&
113 pos.x + hitbox.getSize()[0] <= sizeWindow.first) {
114 newPosX += speedValue;
115 }
116 }
117
118 if (keyboard.isKeyIsPressed(VirtualKeyBoard::CONTROL::LEFT)) {
119 if (!hitbox.isCollisionInDirection(Hitbox::DIRECTION::LEFT) &&
120 pos.x >= 0) {
121 newPosX -= speedValue;
122 }
123 }
124
125 if (keyboard.isKeyIsPressed(VirtualKeyBoard::CONTROL::UP)) {
126 if (!hitbox.isCollisionInDirection(Hitbox::DIRECTION::UP) &&
127 pos.y >= 0) {
128 newPosY -= speedValue;
129 }
130 }
131
132 if (newPosX != 0.0f && newPosY != 0.0f) {
133 float length = std::sqrt(newPosX * newPosX + newPosY * newPosY);
134 newPosX /= length;
135 newPosY /= length;
136 }
137 pos.x += newPosX;
138 pos.y += newPosY;
139 movePlayer.restart();
140}
