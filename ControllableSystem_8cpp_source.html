10int Systems::controllableSystem(World &world)
11{
12 Registry &r = world.getRegistry();
13 CreateEntity &ce = world.getClassCreateEntity();
14 const std::pair<std::size_t, std::size_t> &sizeWindow = world.getSizeWindow();
15 ComponentArray<Position> &position = r.get_components<Position>();
16 ComponentArray<Controllable> &controllables = r.get_components<Controllable>();
17 ComponentArray<Clock> &clocks = r.get_components<Clock>();
18 ComponentArray<Hitbox> &hitboxs = r.get_components<Hitbox>();
19 ComponentArray<Force> &forces = r.get_components<Force>();
20 ComponentArray<Area> &areas = r.get_components<Area>();
21 ComponentArray<Drawable::Drawable> &drawables = r.get_components<Drawable::Drawable>();
22
23 std::size_t index = 0;
24
25 for (std::optional<Controllable> &controllable : controllables) {
26 if (!controllable.has_value() || index >= position.size() || !position[index].has_value() ||
27 index >= clocks.size() || !clocks[index].has_value())
28 {
29 index += 1;
30 continue;
31 }
32 VirtualKeyBoard &keyBoardControllable = controllable->getVirtualKeyboard();
33 if (index < hitboxs.size() && hitboxs[index].has_value()) {
34 movePlayerFromControllable(
35 position[index]->getPosition(), hitboxs[index].value(),
36 keyBoardControllable,
37 clocks[index].value(), sizeWindow
38 );
39 position[index]->setPosition(position[index]->getPosition());
40 }
41 int indexForce = isForceEquipped(forces, index);
42 if (indexForce != -1) {
43 shootHandling(
44 ce, r, keyBoardControllable,
45 position[index].value(), clocks[index].value(), index,
46 forces[indexForce].value()
47 );
48 } else {
49 shootHandling(
50 ce, r, keyBoardControllable,
51 position[index].value(), clocks[index].value(), index
52 );
53 }
54 if (index < areas.size() && areas[index].has_value()) {
55 forceHandling(
56 keyBoardControllable, clocks[index].value(), areas[index].value(),
57 forces, position, index, drawables
58 );
59 }
60 index += 1;
61 }
62 return (0);
63}

65void Systems::shootHandling(
66 CreateEntity &ce, Registry &r, VirtualKeyBoard &keyboard, Position &currentPosition,
67 Clock &clocks, const std::size_t &index, Force &forces
68)
69{
70 static constexpr float SPAM_PROTECTION_DELAY = 0.1f;
71 static constexpr float CHARGE_THRESHOLD = 0.3f;
72 static constexpr float MAX_CHARGE_TIME = 1.1f;
73 static constexpr float BASE_DAMAGE = 80.0f;
74
75 sf::Clock &keyPressing = clocks.getClock("keyPressingShoot");
76 sf::Clock &spamProtection = clocks.getClock("spamProtectionShoot");
77
78 if (spamProtection.getElapsedTime().asSeconds() < SPAM_PROTECTION_DELAY) {
79 return;
80 }
81
82 if (keyboard.isKeyIsPressed(VirtualKeyBoard::CONTROL::SHOOT)) {
83 if (!keyboard.isChargingShoot()) {
84 keyboard.setIsChargingShoot(true);
85 keyPressing.restart();
86 }
87 return;
88 }
89
90 if (!keyboard.isChargingShoot()) {
91 forces.stopShoot();
92 return;
93 }
94
95 forces.haveToShoot();
96 keyboard.setIsChargingShoot(false);
97 const float chargeTime = keyPressing.getElapsedTime().asSeconds();
98
99 if (chargeTime >= CHARGE_THRESHOLD) {
100 const float clampedTime = std::min(chargeTime, MAX_CHARGE_TIME);
101 const float damage = BASE_DAMAGE * (clampedTime / MAX_CHARGE_TIME);
102 ce.createBeam(currentPosition.getPosition(), r, damage, index, true, true);
103 } else {
104 ce.createLaser(currentPosition.getPosition(), r, index, true, true);
105 }
106
107 spamProtection.restart();
108}

153void Systems::movePlayerFromControllable(
154 sf::Vector2f &pos, const Hitbox &hitbox, VirtualKeyBoard &keyboard, Clock &clock,
155 const std::pair<std::size_t, std::size_t> &sizeWindow
156)
157{
158 sf::Clock &movePlayer = clock.getClock("movePlayer");
159 if (movePlayer.getElapsedTime().asMilliseconds() <= 16) {
160 return;
161 }
162
163 static const float speedValue = 15; // a changer en fonction de la vitesse
164
165 float newPosX = 0.0f;
166 float newPosY = 0.0f;
167
168 if (keyboard.isKeyIsPressed(VirtualKeyBoard::CONTROL::DOWN)) {
169 if (!hitbox.isCollisionInDirection(Hitbox::DIRECTION::DOWN) &&
170 pos.y + hitbox.getSize()[1] <= sizeWindow.second) {
171 newPosY += speedValue;
172 }
173 }
174
175 if (keyboard.isKeyIsPressed(VirtualKeyBoard::CONTROL::RIGHT)) {
176 if (!hitbox.isCollisionInDirection(Hitbox::DIRECTION::RIGHT) &&
177 pos.x + hitbox.getSize()[0] <= sizeWindow.first) {
178 newPosX += speedValue;
179 }
180 }
181
182 if (keyboard.isKeyIsPressed(VirtualKeyBoard::CONTROL::LEFT)) {
183 if (!hitbox.isCollisionInDirection(Hitbox::DIRECTION::LEFT) &&
184 pos.x >= 0) {
185 newPosX -= speedValue;
186 }
187 }
188
189 if (keyboard.isKeyIsPressed(VirtualKeyBoard::CONTROL::UP)) {
190 if (!hitbox.isCollisionInDirection(Hitbox::DIRECTION::UP) &&
191 pos.y >= 0) {
192 newPosY -= speedValue;
193 }
194 }
195
196 if (newPosX != 0.0f && newPosY != 0.0f) {
197 float scale = speedValue / (std::sqrt(newPosX * newPosX + newPosY * newPosY));
198 newPosX *= scale;
199 newPosY *= scale;
200 }
201 pos.x += newPosX;
202 pos.y += newPosY;
203 movePlayer.restart();
204}

267void Systems::equipedForce(
268 const Area &areaControllable, ComponentArray<Force> &forces, ComponentArray<Position> &position,
269 const std::size_t &idBelong, ComponentArray<Drawable::Drawable> &drawable
270)
271{
272 std::size_t index = 0;
273
274 for (std::optional<Force> &force : forces) {
275 if (!force || !force.has_value()) {
276 index += 1;
277 continue;
278 }
279 if (!force->isEquipped() && force->getIdBelongForce() >= 0) {
280 force->reattachedForce();
281 return;
282 }
283 if (force->isEquipped()) {
284 force->unEquipped();
285 return;
286 }
287 if (areaControllable.isInAreaOfEntity(index)) {
288 float sizeForce = 0.0;
289 float sizeControllable = 0.0;
290 if (index < drawable.size() && idBelong < drawable.size() && drawable[index].has_value() && drawable[idBelong].has_value()) {
291 sizeForce = (drawable[index]->getCurrentRect()._rectWidth * drawable[index]->getScale().x) / 2;
292 sizeControllable = (drawable[idBelong]->getCurrentRect()._rectWidth * drawable[idBelong]->getScale().x) / 2;
293 }
294 force->equipped(idBelong, isToTheRight(position, idBelong, index, sizeForce, sizeControllable));
295 return;
296 }
297 index += 1;
298 }

312int Systems::isForceEquipped(ComponentArray<Force> &forces, const std::size_t &index)
313{
314 int indexEntity = 0;
315
316 for (std::optional<Force> &force : forces) {
317 if (!force || !force.has_value()) {
318 indexEntity++;
319 continue;
320 }
321 if (force->isEquipped() && force->getIdBelongForce() == index) {

324 indexEntity++;
