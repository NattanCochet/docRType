75void Assets::loadSprite(std::string filepath, sf::IntRect rect, std::string name, sf::Vector2f scale)
76{
77 sf::Texture *texture = new sf::Texture();
78 sf::Sprite sprite;
79 if (texture->loadFromFile(filepath, rect) == false)
80 throw Error(Error::MISTAKE::SPRITE_ERROR);
81 sprite.setTexture(*texture);
82 sf::FloatRect middle = sprite.getLocalBounds();
83 sprite.setOrigin(middle.width / 2, middle.height / 2);
84 sprite.setScale(scale);
85 _mapSprites.insert(std::make_pair(name, sprite));
86}

88void Assets::loadText(std::string filepath, std::string name)
89{
90 sf::Font *textFont = new sf::Font();
91 if (textFont->loadFromFile(filepath) == false)
92 throw Error(Error::MISTAKE::FONT_ERROR);
93 sf::Text text("", *textFont, 0);
94 _mapTexts.insert(std::make_pair(name, text));
95}
