82void Systems::localAnimations(size_t index, std::optional<Drawable> &drawable, World &world)
83{
84 sf::Clock *clock = drawable->getClock();
85 sf::Time timeSpend = clock->getElapsedTime();
86 float seconds = timeSpend.asSeconds();
87 sf::IntRect *rect = drawable->getRect();
88 std::string name = drawable->getPath();
89 Registry &r = world.getRegistry();
90 if (name == "background" && seconds > 0.08) {
91 if (rect->left >= 3179)
92 drawable->setRect({0, rect->top, rect->width, rect->height});
93 else
94 drawable->setRect({rect->left + 2, rect->top, rect->width, rect->height});
95 drawable->restartClock();
96 }
97 if (name == "explosion" && seconds > 0.1) {
98 if (rect->left >= 216 - 36)
99 r.kill_entity(r.entity_from_index(index));
100 else
101 drawable->setRect({rect->left + 36, rect->top, rect->width, rect->height});

183void Systems::changeText(sf::Keyboard::Key key, std::optional<Drawable> &drawable)
184{
185 std::string message = drawable->getMessage();
186 if (key < 26) {
187 char c = key + 'a';
188 drawable->setMessage(message + c);
189 }
190 if (key >= 26 && key <= 35) {
191 char c = key - 26 + '0';
192 drawable->setMessage(message + c);
193 }
194 if (key >= 75 && key <= 84) {
195 char c = key - 75 + '0';
196 drawable->setMessage(message + c);
197 }
198 switch (key) {
199 case sf::Keyboard::Space:
200 drawable->setMessage(message + " ");
201 break;
202 case sf::Keyboard::BackSpace:
203 if (message.size() > 0)
204 message.pop_back();
205 drawable->setMessage(message);
206 break;
207 default:
208 break;
209 }
210}
