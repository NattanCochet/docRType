10int Systems::controllableSystem(World &world, NetworkServer &server)
11{
12 Registry &r = world.getRegistry();
13 CreateEntity &ce = world.getClassCreateEntity();
14 const std::pair<std::size_t, std::size_t> &sizeWindow = world.getSizeWindow();
15 ComponentArray<Position> &position = r.get_components<Position>();
16 ComponentArray<Controllable> &controllables = r.get_components<Controllable>();
17 ComponentArray<Clock> &clocks = r.get_components<Clock>();
18 ComponentArray<Hitbox::Hitbox> &hitboxs = r.get_components<Hitbox::Hitbox>();
19 ComponentArray<Force> &forces = r.get_components<Force>();
20 ComponentArray<Area> &areas = r.get_components<Area>();
21 ComponentArray<Drawable::Drawable> &drawables = r.get_components<Drawable::Drawable>();
22
23 std::size_t index = 0;
24
25 for (std::optional<Controllable> &controllable : controllables) {
26 if (!controllable.has_value() || index >= position.size() || !position[index].has_value() ||
27 index >= clocks.size() || !clocks[index].has_value()) {
28 index += 1;
29 continue;
30 }
31 VirtualKeyBoard &keyBoardControllable = controllable->getVirtualKeyboard();
32 if (index < hitboxs.size() && hitboxs[index].has_value()) {
33 movePlayerFromControllable(position[index]->getPosition(), hitboxs[index].value(), keyBoardControllable,
34 clocks[index].value(), sizeWindow, controllable->getSpeed());
35 position[index]->setPosition(position[index]->getPosition());
36 }
37 int indexForce = isForceEquipped(forces, index);
38 if (indexForce != -1) {
39 shootHandling(ce, r, keyBoardControllable, position[index].value(), clocks[index].value(), index,
40 forces[indexForce].value());
41 } else {
42 shootHandling(ce, r, keyBoardControllable, position[index].value(), clocks[index].value(), index);
43 }
44 if (index < areas.size() && areas[index].has_value()) {
45 forceHandling(
46 keyBoardControllable, clocks[index].value(), areas[index].value(), forces, position, index, drawables);
47 }
48 index += 1;
49 }
50 return (0);
51}

53void Systems::shootHandling(CreateEntity &ce, Registry &r, VirtualKeyBoard &keyboard, Position &currentPosition,
54 Clock &clocks, const std::size_t &index, Force &forces)
55{
56 static constexpr float SPAM_PROTECTION_DELAY = 0.1f;
57 static constexpr float CHARGE_THRESHOLD = 0.3f;
58 static constexpr float MAX_CHARGE_TIME = 1.1f;
59 static constexpr float BASE_DAMAGE = 80.0f;
60
61 sf::Clock &keyPressing = clocks.getClock("keyPressingShoot");
62 sf::Clock &spamProtection = clocks.getClock("spamProtectionShoot");
63
64 if (spamProtection.getElapsedTime().asSeconds() < SPAM_PROTECTION_DELAY) {
65 return;
66 }
67
68 if (keyboard.isKeyIsPressed(VirtualKeyBoard::CONTROL::SHOOT)) {
69 if (!keyboard.isChargingShoot()) {
70 keyboard.setIsChargingShoot(true);
71 forces.stopShoot();
72 keyPressing.restart();
73 }
74 return;
75 }
76
77 if (!keyboard.isChargingShoot()) {
78 forces.stopShoot();
79 return;
80 }
81
82 forces.haveToShoot();
83 keyboard.setIsChargingShoot(false);
84 const float chargeTime = keyPressing.getElapsedTime().asSeconds();
85
86 if (chargeTime >= CHARGE_THRESHOLD) {
87 const float clampedTime = std::min(chargeTime, MAX_CHARGE_TIME);
88 const float damage = BASE_DAMAGE * (clampedTime / MAX_CHARGE_TIME);
89 ce.createBeam(currentPosition.getPosition(), r, damage, index);
90 } else {
91 ce.createShootPlayer(currentPosition.getPosition(), r, index);
92 }
93
94 spamProtection.restart();
95}

138void Systems::movePlayerFromControllable(sf::Vector2f &pos, const Hitbox::Hitbox &hitbox, VirtualKeyBoard &keyboard,
139 Clock &clock, const std::pair<std::size_t, std::size_t> &sizeWindow, const int speed)
140{
141 sf::Clock &movePlayer = clock.getClock("movePlayer");
142 if (movePlayer.getElapsedTime().asMilliseconds() <= 16) {
143 return;
144 }
145
146 float newPosX = 0.0f;
147 float newPosY = 0.0f;
148
149 if (keyboard.isKeyIsPressed(VirtualKeyBoard::CONTROL::DOWN)) {
150 if (!hitbox.isCollisionInDirection(Hitbox::DIRECTION::DOWN) &&
151 pos.y + hitbox.getSizeHitbox(0).y <= sizeWindow.second) {
152 newPosY += speed;
153 }
154 }
155
156 if (keyboard.isKeyIsPressed(VirtualKeyBoard::CONTROL::RIGHT)) {
157 if (!hitbox.isCollisionInDirection(Hitbox::DIRECTION::RIGHT) &&
158 pos.x + hitbox.getSizeHitbox(0).x <= sizeWindow.first) {
159 newPosX += speed;
160 }
161 }
162
163 if (keyboard.isKeyIsPressed(VirtualKeyBoard::CONTROL::LEFT)) {
164 if (!hitbox.isCollisionInDirection(Hitbox::DIRECTION::LEFT) && pos.x >= 0) {
165 newPosX -= speed;
166 }
167 }
168
169 if (keyboard.isKeyIsPressed(VirtualKeyBoard::CONTROL::UP)) {
170 if (!hitbox.isCollisionInDirection(Hitbox::DIRECTION::UP) && pos.y >= 0) {
171 newPosY -= speed;
172 }
173 }
174
175 if (newPosX != 0.0f && newPosY != 0.0f) {
176 float scale = speed / (std::sqrt(newPosX * newPosX + newPosY * newPosY));
177 newPosX *= scale;
178 newPosY *= scale;
179 }
180 pos.x += newPosX;
181 pos.y += newPosY;
182 movePlayer.restart();
183}

245void Systems::equipedForce(const Area &areaControllable, ComponentArray<Force> &forces,
246 ComponentArray<Position> &position, const std::size_t &idBelong, ComponentArray<Drawable::Drawable> &drawable)
247{
248 std::size_t index = 0;
249
250 for (std::optional<Force> &force : forces) {
251 if (!force || !force.has_value()) {
252 index += 1;
253 continue;
254 }
255 if (!force->isEquipped() && force->getIdBelongForce() >= 0) {
256 force->reattachedForce();
257 return;
258 }
259 if (force->isEquipped()) {
260 force->unEquipped();
261 return;
262 }
263 if (areaControllable.isInAreaOfEntity(index)) {
264 float sizeForce = 0.0;
265 float sizeControllable = 0.0;
266 if (index < drawable.size() && idBelong < drawable.size() && drawable[index].has_value() &&
267 drawable[idBelong].has_value()) {
268 sizeForce = (drawable[index]->getCurrentRect()._rectWidth * drawable[index]->getScale().x) / 2;
269 sizeControllable =
270 (drawable[idBelong]->getCurrentRect()._rectWidth * drawable[idBelong]->getScale().x) / 2;
271 }
272 force->equipped(idBelong, isToTheRight(position, idBelong, index, sizeForce, sizeControllable));
273 return;
274 }
275 index += 1;
276 }

288int Systems::isForceEquipped(ComponentArray<Force> &forces, const std::size_t &index)
289{
290 int indexEntity = 0;
291
292 for (std::optional<Force> &force : forces) {
293 if (!force || !force.has_value()) {
294 indexEntity++;
295 continue;
296 }
297 if (force->isEquipped() && force->getIdBelongForce() == index) {
298 return (indexEntity);

300 indexEntity++;
301 }
