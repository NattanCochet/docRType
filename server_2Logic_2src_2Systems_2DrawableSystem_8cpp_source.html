17 std::size_t index = 0;

19 for (std::optional<Drawable::Drawable> &drawable : drawables) {

21 index += 1;
22 continue;

28 drawable->setDirectionFrame(0);
29 } else if (y_axis == 1) {
30 drawable->setDirectionFrame(3);
31 } else if (y_axis == -1) {
32 drawable->setDirectionFrame(1);
33 } else if (x_axis == 1) {
34 drawable->setDirectionFrame(2);
35 } else if (x_axis == -1) {
36 drawable->setDirectionFrame(4);
37 }
38 } else {
39 drawable->setDirectionFrame(0);
40 }
41 sf::Clock &clock = drawable->getClock();
42 if (clock.getElapsedTime().asSeconds() >= drawable->getSecondDependOnDirection()) {
43 drawable->passNextFrame();
44 clock.restart();
45 }
46 index += 1;
47 }
48 return (0);

52{

63 x -= 1;
