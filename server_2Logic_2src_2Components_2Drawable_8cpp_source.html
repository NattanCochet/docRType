200void Drawable::Drawable::passNextFrame() noexcept
201{
202 this->_currentFrame += 1;
203 switch (this->_directionFrame) {
204 case 0:
205 this->_currentFrame %= this->_defaultFrame.size();
206 break;
207 case 1:
208 this->_currentFrame %= this->_frameUP.size();
209 break;
210 case 2:

216 case 4:

218 break;
219
220 default:
221 this->_currentFrame %= this->_defaultFrame.size();
222 break;

227{
