473 void CalculateFractalBounding()
474 {
475 float gain = FastAbs(mGain);
476 float amp = gain;
477 float ampFractal = 1.0f;
478 for (int i = 1; i < mOctaves; i++)
479 {
480 ampFractal += amp;
481 amp *= gain;
482 }
483 mFractalBounding = 1 / ampFractal;
484 }

661 // Noise Coordinate Transforms (frequency, and possible skew or rotation)
662
663 template <typename FNfloat>
664 void TransformNoiseCoordinate(FNfloat& x, FNfloat& y) const
665 {
666 x *= mFrequency;
667 y *= mFrequency;
668
669 switch (mNoiseType)
670 {
671 case NoiseType_OpenSimplex2:
672 case NoiseType_OpenSimplex2S:
673 {
674 const FNfloat SQRT3 = (FNfloat)1.7320508075688772935274463415059;
675 const FNfloat F2 = 0.5f * (SQRT3 - 1);
676 FNfloat t = (x + y) * F2;
677 x += t;
678 y += t;
679 }
680 break;
681 default:
682 break;
683 }
684 }
685
686 template <typename FNfloat>
687 void TransformNoiseCoordinate(FNfloat& x, FNfloat& y, FNfloat& z) const
688 {
689 x *= mFrequency;
690 y *= mFrequency;
691 z *= mFrequency;
692
693 switch (mTransformType3D)
694 {
695 case TransformType3D_ImproveXYPlanes:
696 {
697 FNfloat xy = x + y;
698 FNfloat s2 = xy * -(FNfloat)0.211324865405187;
699 z *= (FNfloat)0.577350269189626;
700 x += s2 - z;
701 y = y + s2 - z;
702 z += xy * (FNfloat)0.577350269189626;
703 }
704 break;
705 case TransformType3D_ImproveXZPlanes:
706 {
707 FNfloat xz = x + z;
708 FNfloat s2 = xz * -(FNfloat)0.211324865405187;
709 y *= (FNfloat)0.577350269189626;
710 x += s2 - y;
711 z += s2 - y;
712 y += xz * (FNfloat)0.577350269189626;
713 }
714 break;
715 case TransformType3D_DefaultOpenSimplex2:
716 {
717 const FNfloat R3 = (FNfloat)(2.0 / 3.0);
718 FNfloat r = (x + y + z) * R3; // Rotation, not skew
719 x = r - x;
720 y = r - y;
721 z = r - z;
722 }
723 break;
724 default:
725 break;
726 }
727 }

730 {
731 switch (mRotationType3D)
732 {

735 break;
736 case RotationType3D_ImproveXZPlanes:

738 break;
739 default:

745 break;

749 }

1007 j *= PrimeY;
