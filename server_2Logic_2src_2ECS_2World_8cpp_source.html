359bool World::spawnEntitiesInWorld(std::size_t level, std::size_t spawnAfterWindow, int sizeWorld)
360{
361 if (_isLevelFinishedToSpawn) {
362 return (true);
363 }
364
365 std::size_t seed = this->getSeedForLevel(level);
366
367 if (_spawnAfterWindow + spawnAfterWindow >= sizeWorld) {
368 _isLevelFinishedToSpawn = true;
369 spawnAfterWindow = sizeWorld - _spawnAfterWindow;
370 }
371 std::pair<int, int> x_axis = std::make_pair(_spawnAfterWindow, _spawnAfterWindow + spawnAfterWindow);
372 std::pair<int, int> y_axis = std::make_pair(0, _heightWindow);
373 std::vector<std::vector<int>> world = this->generateWorldWithID(level, x_axis, y_axis);
374 _spawnAfterWindow += spawnAfterWindow;
375
376 for (std::size_t y = 0; y < spawnAfterWindow; y++) {
377 for (std::size_t x = 0; x < _heightWindow; x++) {
378 SpawnRule::EntityType entity;
379 try {
380 entity = _srg.getEntity(world[y][x]);
381 }
382 catch (ErrorKeyNotFound &e) {
383 std::cout << e.what() << std::endl;
384 continue;
385 }
386 if (entity.id != 0 && _srg.hasEnoughtPlaceForEnemy(world, y, x, entity.sizeEntity)) {
387 std::size_t indexNewEntityAfterDead = this->generateRandomValueForEntityAfterDead(seed);
388 std::cout << "je fais spawn avec id = " << entity.id << ", avec sad = " << indexNewEntityAfterDead
389 << std::endl;
390 this->spawnEntityFromGeneratedWorld(entity.id, _widthWindow + y, x, indexNewEntityAfterDead);
391 }
392 }
393 }
394
395 if (_isLevelFinishedToSpawn) {
396 if (seed % 2 == 0) {
397 _bossIDAtTheEnd = seed % 3; // pour 3 boss;
398 this->spawnEntityFromGeneratedWorld(
399 15 + _bossIDAtTheEnd, _widthWindow + spawnAfterWindow + 20, _heightWindow / 2);
400 } else {
401 _bossIDAtTheEnd = -1;
402 }
403 }
404
405 return (_isLevelFinishedToSpawn);
406}

423void World::resetFinishedLevel(bool isWin, NetworkServer &server)
424{
425 this->_isLevelFinishedToSpawn = false;
426 this->_isLevelFinished = false;
427 this->_spawnAfterWindow = _widthWindow;
428 this->_stateWorld = NOTHING;
429 this->_bossIDAtTheEnd = -1;
430 this->_isForceInWorld = false;
431 if (isWin) {
432 if (this->_currentLevel == this->_maxLevel - 1) {
433 this->generateNextSeedForNextLevel();
434 }
435 this->_currentLevel += 1;
436 }
437 for (std::pair<const std::size_t, ClientInWorld> &eachClient : this->_idClients) {
438 eachClient.second.saveScore();

440 endMessage.header.id = Protocol::SERVER_GAME_END;
441 endMessage << isWin;
442 endMessage << eachClient.second.getScore();
443 server.SendToClient(endMessage, eachClient.first);
444 eachClient.second.resetScore();
