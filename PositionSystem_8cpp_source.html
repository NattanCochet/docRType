10int Systems::positionSystem(World &world)
11{
12 Registry &r = world.getRegistry();
13 ComponentArray<Position> &positions = r.get_components<Position>();

16 std::size_t sizeArrayPositions = positions.size();

18 const auto isCircleInWindow = [windowWidth, windowHeight](const sf::Vector2f &pos, const Vector<int, 2> &size) {
19 const int radius = size[0];
20 return (pos.x + radius >= 0 && pos.x - radius <= windowWidth) &&
21 (pos.y + radius >= 0 && pos.y - radius <= windowHeight);
22 };
23
24 for (std::size_t index = 0; index < sizeArrayPositions; index++) {
25 std::optional<Position> &position = positions[index];
26 std::optional<Hitbox> &hitbox = hitboxs[index];
27
28 if (!position.has_value() || !hitbox.has_value()) {
29 continue;
30 }
31
32 const sf::Vector2f &pos = position->getPosition();
33 const Vector<int, 2> &sizeHitbox = hitbox->getSize();
34 bool isInWindow = false;
35
36 switch (hitbox->getType()) {
37 case Hitbox::TYPE::RECTANGLE:
38 isInWindow = isRectangleInWindow(windowWidth, windowHeight, pos, sizeHitbox, hitbox->getRotation());
39 break;
40 case Hitbox::TYPE::CIRCULAR:
41 isInWindow = isCircleInWindow(pos, sizeHitbox);
42 break;
43 default:
44 isInWindow = false;
45 break;
46 }
47 if (!position->getIsAppearOnTheWindow() && isInWindow) {
48 position->setIsAppearOnTheWindow(true);
49 } else if (position->getIsAppearOnTheWindow() && !isInWindow) {
50 r.kill_entity(r.entity_from_index(index));
51 }
52 }
53 return (0);

56bool Systems::isRectangleInWindow(const std::size_t &windowWidth, const std::size_t &windowHeight, const sf::Vector2f &pos, const Vector<int, 2> &size, const float &rotation)
57{
58 float rad = rotation * M_PI / 180.0f;
59 float cos_r = std::cos(rad);
60 float sin_r = std::sin(rad);
61
62 std::array<sf::Vector2f, 4> corners = {
63 sf::Vector2f(0, 0),
64 sf::Vector2f(size[0], 0),
65 sf::Vector2f(size[0], size[1]),
66 sf::Vector2f(0, size[1])
67 };
68
69 for (auto& corner : corners) {
70 float x_rot = corner.x * cos_r - corner.y * sin_r;
71 float y_rot = corner.x * sin_r + corner.y * cos_r;
72
73 corner.x = x_rot + pos.x;
74 corner.y = y_rot + pos.y;
75 }
76
77 for (const auto& corner : corners) {
78 if (corner.x >= 0 && corner.x <= windowWidth &&
79 corner.y >= 0 && corner.y <= windowHeight) {
80 return (true);
81 }
82 }
83
84 return (false);
