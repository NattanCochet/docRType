69void Buttons::createRoomButton(std::size_t index)
70{
71 Registry &r = _world->getRegistry();
72 std::map<std::string, int> &keys = _world->getKeysIndex();
73 ComponentArray<Drawable> &drawables = r.get_components<Drawable>();
74 ComponentArray<Clickable> &clickables = r.get_components<Clickable>();
75 std::string roomName = drawables[keys["roomNameCreate"]]->getMessage();
76 std::string roomPassword = drawables[keys["roomPasswordCreate"]]->getMessage();
77 std::string roomMaxStr = drawables[keys["roomMaxNbrPlayerCreate"]]->getMessage();
78 bool isPublic = true;
79 void *buttonT = clickables[keys["roomIsPublicCreate"]]->getInfos();
80 if (roomName.empty()) {
81 drawables[keys["roomErrorCreate"]]->setMessage("Room must have a name");
82 return;
83 }
84 if (buttonT && clickables[keys["roomIsPublicCreate"]]->getKindButton() == Clickable::SPLITBUTTON) {
85 Buttons::SplitButton_t *button = static_cast<Buttons::SplitButton_t *>(buttonT);
86 isPublic = button->_active;
87 } else {
88 drawables[keys["roomErrorCreate"]]->setMessage("Public button error");
89 return;
90 }
91 if (roomPassword.empty() && !isPublic) {
92 drawables[keys["roomErrorCreate"]]->setMessage("Private room must have a password");
93 return;
94 }
95 if (roomMaxStr.empty()) {
96 drawables[keys["roomErrorCreate"]]->setMessage("Must set the maximum number of players");
97 return;
98 }
99 std::size_t maxPlayers = 0;
100 std::stringstream stream(roomMaxStr);
101 stream >> maxPlayers;
102 if (maxPlayers < 1) {
103 drawables[keys["roomErrorCreate"]]->setMessage("Maximum number of players must be at least 1");
104 return;
105 }
106 _network->sendCreateRoom(_world->getClient(), roomName, roomPassword, isPublic, maxPlayers);
107 _world->setRoomName(roomName);
108 _world->setNbInRoom({1, maxPlayers});
109}

208 Buttons::SplitButton_t *button = static_cast<Buttons::SplitButton_t *>(buttonT);
209 if (button->_active) {
210 return;
211 }
212 button->_active = true;
213 if (button->_activePt != nullptr) {
214 *button->_activePt = true;
215 }
216 Buttons::SplitButton_t *other = button->_other;
217 other->_active = false;
218 if (other->_activePt != nullptr) {
219 *other->_activePt = false;
220 }
221 if (button->_actionActivation) {
222 button->_actionActivation(button->_actionActivationParam);
223 }
224 drawables[button->_index]->setColor(button->_activerColor);
225 drawables[other->_index]->setColor(other->_defaultColor);

335bool Buttons::writeKeyButton(sf::Keyboard::Key key, World &world, std::size_t index)
336{
337 ComponentArray<Drawable> &drawables = world.getRegistry().get_components<Drawable>();
338
339 std::string message = "";
340 if (key == sf::Keyboard::Unknown) {
341 drawables[index]->setMessage("EMPTY");
342 return false;
343 }
344 if (key < 26) {
345 char c = key + 'A';
346 drawables[index]->setMessage(message + c);
347 return true;
348 }
349 if (key >= 26 && key <= 35) {
350 char c = key - 26 + '0';
351 drawables[index]->setMessage(message + c);
352 return true;
353 }
354 if (key >= 75 && key <= 84) {
355 char c = key - 75 + '0';
356 drawables[index]->setMessage(message + c);
357 return true;
358 }
359 if (key == sf::Keyboard::Space) {
360 drawables[index]->setMessage("SPACE");
361 return true;
362 }
363 if (key == sf::Keyboard::Tab) {
364 drawables[index]->setMessage("TAB");
365 return true;
366 }
367 if (key == sf::Keyboard::LAlt) {
368 drawables[index]->setMessage("L ALT");
369 return true;
370 }
371 if (key == sf::Keyboard::LControl) {
372 drawables[index]->setMessage("L CTRL");
373 return true;
374 }
375 if (key == sf::Keyboard::LShift) {
376 drawables[index]->setMessage("L SHIFT");
377 return true;
378 }
379 if (key == sf::Keyboard::RAlt) {
380 drawables[index]->setMessage("R ALT");
381 return true;
382 }
383 if (key == sf::Keyboard::RControl) {
384 drawables[index]->setMessage("R CTRL");
385 return true;
386 }
387 if (key == sf::Keyboard::RShift) {
388 drawables[index]->setMessage("R SHIFT");
389 return true;
390 }
391 if (key == sf::Keyboard::Enter) {
392 drawables[index]->setMessage("ENTER");
393 return true;
394 }
395 if (key == sf::Keyboard::Left) {
396 drawables[index]->setMessage("LEFT ARROW");
397 return true;
398 }
399 if (key == sf::Keyboard::Right) {
400 drawables[index]->setMessage("RIGHT ARROW");
401 return true;
402 }
403 if (key == sf::Keyboard::Up) {
404 drawables[index]->setMessage("UP ARROW");
405 return true;
406 }
407 if (key == sf::Keyboard::Down) {
408 drawables[index]->setMessage("DOWN ARROW");
409 return true;
410 }
411 drawables[index]->setMessage("EMPTY");
412 return false;
