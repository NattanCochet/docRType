357bool World::spawnEntitiesInWorld(std::size_t level, std::size_t spawnAfterWindow, int sizeWorld)
358{
359 if (_isLevelFinishedToSpawn) {
360 return (true);
361 }
362 if (_spawnAfterWindow + spawnAfterWindow >= sizeWorld) {
363 _isLevelFinishedToSpawn = true;
364 spawnAfterWindow = sizeWorld - _spawnAfterWindow;
365 }
366 std::pair<int, int> x_axis = std::make_pair(_spawnAfterWindow, _spawnAfterWindow + spawnAfterWindow);
367 std::pair<int, int> y_axis = std::make_pair(0, _heightWindow);
368 std::vector<std::vector<int>> world = this->generateWorldWithID(level, x_axis, y_axis);
369 _spawnAfterWindow += spawnAfterWindow;
370
371 for (std::size_t y = 0; y < spawnAfterWindow; y++) {
372 for (std::size_t x = 0; x < _heightWindow; x++) {
373 SpawnRule::EntityType entity;
374 try {
375 entity = _srg.getEntity(world[y][x]);
376 } catch (ErrorKeyNotFound &e) {
377 std::cout << e.what() << std::endl;
378 continue;
379 }
380 if (entity.id != 0 && _srg.hasEnoughtPlaceForEnemy(world, y, x, entity.sizeEntity)) {
381 std::cout << "je fais spawn avec id = " << entity.id << std::endl;
382 this->spawnEntityFromGeneratedWorld(entity.id, _widthWindow + y, x, 0);
383 }
384 }
385 }
386
387 if (_isLevelFinishedToSpawn) {
388 std::size_t seed = this->getSeedForLevel(level);
389
390 if (seed % 2 == 0) {
391 _bossIDAtTheEnd = seed % 1; // pour 4 boss par exemple;
392 } else {
393 _bossIDAtTheEnd = -1;
394 }
395 }
396
397 return (_isLevelFinishedToSpawn);
398}
