108std::size_t World::addClientInWorld(std::size_t clientID, std::string password)
109{
110 if (!_isPublic) {

112 throw ErrorPasswordIncorrect(password);
113 }
114 }
115 if (this->isWorldFull()) {
116 throw ErrorWorldFull(_nbrPlayerMax);
117 }
118 if (this->isClientIDPresentAsBool(clientID)) {
119 throw (ErrorClientAlreadyPresentInWorld(clientID));
120 }
121 _idClients.push_back(clientID);
122 return (_idClients.size());
123}

129std::size_t World::deleteClientInWorld(std::size_t clientID)
130{
131 int index = this->getClientIDPresenceAsInt(clientID);

134 throw ErrorVector("ClientID " + std::to_string(clientID) + " not found.");

137 _idClients.erase(_idClients.begin() + index);
