337bool World::spawnEntitiesInWorld(std::size_t level, std::size_t spawnAfterWindow, int sizeWorld)
338{
339 if (_isLevelFinishedToSpawn) {
340 return (true);
341 }
342 if (_spawnAfterWindow + spawnAfterWindow >= sizeWorld) {
343 _isLevelFinishedToSpawn = true;
344 spawnAfterWindow = sizeWorld - _spawnAfterWindow;
345 }
346 std::pair<int, int> x_axis = std::make_pair(_spawnAfterWindow, _spawnAfterWindow + spawnAfterWindow);
347 std::pair<int, int> y_axis = std::make_pair(0, _heightWindow);
348 std::vector<std::vector<int>> world = this->generateWorldWithID(level, x_axis, y_axis);
349 _spawnAfterWindow += spawnAfterWindow;
350
351 for (std::size_t y = 0; y < spawnAfterWindow; y++) {
352 for (std::size_t x = 0; x < _heightWindow; x++) {
353 SpawnRule::EntityType entity;
354 try {
355 entity = _srg.getEntity(world[y][x]);
356 } catch (ErrorKeyNotFound &e) {
357 std::cout << e.what() << std::endl;
358 continue;
359 }
360 if (_srg.hasEnoughtPlaceForEnemy(world, y, x, entity.sizeEntity)) {
361 this->spawnEntityFromGeneratedWorld(entity.id, _widthWindow + y, x);
362 }
363 }
364 }
365
366 return (_isLevelFinishedToSpawn);
367}

463void World::applyFonctionInWorld()
464{
465 if (_stateWorld != PLAY) {
466 if (_stateWorld == HOLD) {
467 _clockGenerationWorld.pause();
468 } else {
469 _clockGenerationWorld.restartAndPause();
470 }
471 return;
472 }
473 _clockGenerationWorld.start();
474 if (_isLevelFinished) {
475 this->resetFinishedLevel();
476 return;
477 }
478 if ((_spawnAfterWindow == _widthWindow) ||
479 (!_isLevelFinishedToSpawn && _clockGenerationWorld.getElapsedTime().asSeconds() >= SPAWN_TIME_THRESHOLD))
480 {
481 this->spawnEntitiesInWorld(_currentLevel, SPAWN_AFTER_WINDOW, SIZE_WORLD);
482 _clockGenerationWorld.restart();
483 }
484 for (const std::function<int (World &)> &func : this->_r.get_systems()) {
485 func(*this);
486 }
