100void Network::receive_string(Message<Protocol> msg)
101{
102 try {
103 uint32_t stringSize;
104 msg >> stringSize;
105 if (msg.body.size() < stringSize) {
106 throw std::runtime_error("Taille de chaîne invalide ou corruption du message !");
107 }
108 std::string receivedString(stringSize, '\0');
109 std::memcpy(&receivedString[0], msg.body.data() + msg.body.size() - stringSize, stringSize);
110 msg.body.resize(msg.body.size() - stringSize);
111 std::cout << "String reçu : " << receivedString << std::endl;
